## Required JavaScript Features

Every presentation must include:

1. **SlidePresentation Class** — Main controller with:
   - Keyboard navigation (arrows, space, page up/down)
   - Touch/swipe support
   - Mouse wheel navigation
   - Progress bar updates
   - Navigation dots

2. **Intersection Observer** — For scroll-triggered animations:
   - Add `.visible` class when slides enter viewport
   - Trigger CSS transitions efficiently

3. **Optional Enhancements** (match to chosen style):
   - Custom cursor with trail
   - Particle system background (canvas)
   - Parallax effects
   - 3D tilt on hover
   - Magnetic buttons
   - Counter animations

4. **Pro Inline Editing** (only if user opted in during Phase 1 — skip entirely if they said No):
   - Edit toggle button (hidden by default, revealed via hover hotzone or `E` key)
   - Image upload with placement dropdown (below / left / right)
   - Image compression (canvas → JPEG for photos, PNG preserved for transparency)
   - Delete image placeholder with auto-unwrap for layout rows
   - IndexedDB fallback storage (handles presentations > 5MB)
   - Add/delete slides
   - See "Pro Inline Editing System" section below

### Hover Control

**Do NOT use CSS `~` sibling selector for hover-based show/hide.** The CSS-only approach fails because `pointer-events: none` on the toggle button breaks the hover chain.

**Required approach: JS-based hover with delay timeout.**

```javascript
const hotzone = document.getElementById('editHotzone');
const toggle = document.getElementById('editToggle');
hotzone.addEventListener('mouseenter', () => toggle.classList.add('show'));
hotzone.addEventListener('mouseleave', () => {
    if (!editor.isEditing) toggle.classList.remove('show');
});
toggle.addEventListener('click', () => editor.toggle());
```

### Image Placement Logic

**Bottom placement** — Append `.img-placeholder` to the end of `.slide-content`:

```javascript
addImagePlaceholder(position = 'bottom') {
    const slide = this.presentation.slides[this.presentation.currentSlide];
    const content = slide.querySelector('.slide-content');
    const ph = document.createElement('div');
    ph.className = 'img-placeholder reveal';
    ph.textContent = '📷 Click or drag to upload image';

    if (position === 'bottom') {
        ph.style.cssText = 'margin-top: clamp(0.5rem, 1vw, 1rem); max-width: 100%; min-height: clamp(80px, 15vh, 180px); width: 100%;';
        content.appendChild(ph);
    } else {
        /* Left or Right: wrap existing content + image in a row */
        const row = document.createElement('div');
        row.className = 'slide-layout-row';
        const textPanel = document.createElement('div');
        textPanel.className = 'text-panel';
        const children = [...content.children].filter(c =>
            !c.classList.contains('slide-layout-row') &&
            !c.classList.contains('slide-footer')
        );
        children.forEach(c => textPanel.appendChild(c));
        if (position === 'left') {
            row.appendChild(ph);
            row.appendChild(textPanel);
        } else {
            row.appendChild(textPanel);
            row.appendChild(ph);
        }
        content.insertBefore(row, content.firstChild);
    }
    this.enableImageUpload();
    this.save();
}
```

### Image Compression on Upload

**Critical: preserve PNG transparency.** PNG/WebP/GIF keep alpha channel; JPEG/others get white background fill.

```javascript
handleImageFile(file, placeholder) {
    this._compressImage(file).then(dataUrl => {
        placeholder.innerHTML = '';
        placeholder.classList.add('has-image');
        const img = document.createElement('img');
        img.src = dataUrl;
        placeholder.appendChild(img);
        /* Add remove-image button */
        const removeBtn = document.createElement('button');
        removeBtn.className = 'img-remove';
        removeBtn.textContent = '×';
        removeBtn.addEventListener('click', (ev) => {
            ev.stopPropagation();
            placeholder.innerHTML = placeholder.dataset.originalText || '📷 Click or drag to upload image';
            placeholder.classList.remove('has-image');
            this.save();
        });
        placeholder.appendChild(removeBtn);
        this.save();
    });
}

_compressImage(file, maxW = 1920, quality = 0.82) {
    return new Promise(resolve => {
        const img = new Image();
        img.onload = () => {
            let w = img.naturalWidth, h = img.naturalHeight;
            if (w > maxW) { h = Math.round(h * maxW / w); w = maxW; }
            const c = document.createElement('canvas');
            c.width = w; c.height = h;
            const ctx = c.getContext('2d');
            /* PNG/WebP/GIF support transparency — keep as PNG; others → JPEG */
            const keepAlpha = /\/(png|webp|gif)$/i.test(file.type);
            if (!keepAlpha) {
                ctx.fillStyle = '#fff';
                ctx.fillRect(0, 0, w, h);
            }
            ctx.drawImage(img, 0, 0, w, h);
            URL.revokeObjectURL(img.src);
            resolve(keepAlpha ? c.toDataURL('image/png') : c.toDataURL('image/jpeg', quality));
        };
        img.src = URL.createObjectURL(file);
    });
}
```

### IndexedDB Storage (Pro)

**localStorage has a ~5MB limit.** Presentations with embedded images easily exceed this. The Pro editor tries localStorage first, then falls back to IndexedDB automatically.

```javascript
/* IndexedDB helpers */
_idbOpen() {
    return new Promise((resolve, reject) => {
        const r = indexedDB.open('slide-editor', 1);
        r.onupgradeneeded = () => r.result.createObjectStore('kv');
        r.onsuccess = () => resolve(r.result);
        r.onerror = () => reject(r.error);
    });
}
_idbPut(k, v) {
    return this._idbOpen().then(db => new Promise((res, rej) => {
        const tx = db.transaction('kv', 'readwrite');
        tx.objectStore('kv').put(v, k);
        tx.oncomplete = () => { db.close(); res(); };
        tx.onerror = () => { db.close(); rej(); };
    }));
}
_idbGet(k) {
    return this._idbOpen().then(db => new Promise((res, rej) => {
        const tx = db.transaction('kv', 'readonly');
        const r = tx.objectStore('kv').get(k);
        r.onsuccess = () => { db.close(); res(r.result); };
        r.onerror = () => { db.close(); rej(); };
    }));
}

/* Save with automatic fallback */
save() {
    const data = {};
    document.querySelectorAll('.slide').forEach((slide, i) => {
        data[`slide-${i}`] = slide.innerHTML;
    });
    const json = JSON.stringify(data);
    const VERSION = 'v1-content';

    try {
        localStorage.setItem('slide-content', json);
        localStorage.setItem('slide-content-version', VERSION);
        localStorage.removeItem('slide-storage');
        this._showSaveStatus(true);
    } catch (e) {
        /* localStorage full — fall back to IndexedDB */
        localStorage.removeItem('slide-content');
        localStorage.setItem('slide-storage', 'idb');
        this._idbPut('slide-content', json)
            .then(() => this._idbPut('slide-content-version', VERSION))
            .then(() => this._showSaveStatus(true))
            .catch(() => this._showSaveStatus(false));
    }
}

_showSaveStatus(ok) {
    const banner = document.getElementById('editBanner');
    if (ok) {
        banner.textContent = 'Saved ✓';
        banner.style.background = '';
    } else {
        banner.textContent = '⚠ Save failed';
        banner.style.background = 'rgba(200,0,0,0.9)';
    }
    banner.classList.add('visible');
    setTimeout(() => {
        banner.textContent = 'Edit mode · Click text to edit · Drag images · Ctrl+S save · E to exit';
        banner.style.background = '';
        if (!this.isEditing) banner.classList.remove('visible');
    }, 1500);
}

/* Load from whichever storage has data */
loadSaved() {
    const CONTENT_VERSION = 'v1-content';
    const mode = localStorage.getItem('slide-storage');

    if (mode === 'idb') {
        /* Async: load from IndexedDB */
        this._idbGet('slide-content-version').then(ver => {
            if (ver !== CONTENT_VERSION) {
                this._idbPut('slide-content', null);
                this._idbPut('slide-content-version', CONTENT_VERSION);
                return null;
            }
            return this._idbGet('slide-content');
        }).then(json => {
            if (!json) return;
            const d = JSON.parse(json);
            document.querySelectorAll('.slide').forEach((s, i) => {
                if (d[`slide-${i}`]) s.innerHTML = d[`slide-${i}`];
            });
        }).catch(() => {});
        return;
    }

    /* Sync: localStorage */
    if (localStorage.getItem('slide-content-version') !== CONTENT_VERSION) {
        localStorage.removeItem('slide-content');
        localStorage.removeItem('slide-position');
        localStorage.setItem('slide-content-version', CONTENT_VERSION);
        return;
    }
    const saved = localStorage.getItem('slide-content');
    if (!saved) return;
    try {
        const data = JSON.parse(saved);
        document.querySelectorAll('.slide').forEach((slide, i) => {
            if (data[`slide-${i}`]) slide.innerHTML = data[`slide-${i}`];
        });
    } catch (e) { /* ignore */ }
}
```

### Version Key Convention

Use a unique version key per presentation to avoid cross-contamination of localStorage data between different presentations:

```javascript
const CONTENT_VERSION = 'v1-<slide-count>slides-<unique-suffix>';
// Example: 'v1-18slides-logo', 'v2-12slides-pitch'
```

Bump the version whenever the HTML structure changes (slides added/removed, major layout changes) to invalidate stale cache.

### Image Processing

```python
from PIL import Image, ImageDraw

# Circular crop (for logos on modern/clean styles)
def crop_circle(input_path, output_path):
    img = Image.open(input_path).convert('RGBA')
    w, h = img.size
    size = min(w, h)
    left, top = (w - size) // 2, (h - size) // 2
    img = img.crop((left, top, left + size, top + size))
    mask = Image.new('L', (size, size), 0)
    ImageDraw.Draw(mask).ellipse([0, 0, size, size], fill=255)
    img.putalpha(mask)
    img.save(output_path, 'PNG')

# Resize (for oversized images that inflate HTML)
def resize_max(input_path, output_path, max_dim=1200):
    img = Image.open(input_path)
    img.thumbnail((max_dim, max_dim), Image.LANCZOS)
    img.save(output_path, quality=85)
```

### Embedding Images for PDF Export

When burning images into HTML for PDF export (images from localStorage are NOT accessible to headless browsers):

```python
from PIL import Image
import base64, io

def compress_and_embed(image_path, max_width=1920, quality=82):
    """Compress image and return base64 data URL string."""
    img = Image.open(image_path)
    if img.mode == 'RGBA':
        bg = Image.new('RGB', img.size, (255, 255, 255))
        bg.paste(img, mask=img.split()[3])
        img = bg
    elif img.mode != 'RGB':
        img = img.convert('RGB')
    if img.width > max_width:
        ratio = max_width / img.width
        img = img.resize((max_width, int(img.height * ratio)), Image.LANCZOS)
    buf = io.BytesIO()
    img.save(buf, format='JPEG', quality=quality)
    b64 = base64.b64encode(buf.getvalue()).decode()
    return f'data:image/jpeg;base64,{b64}'
```
